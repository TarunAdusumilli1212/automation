const app = firebase.initializeApp(firebaseConfig); const db = firebase.firestore(app); // Initialize VM data structure const defaultVMStatus = { "VM1": { status: "Free", user: "" }, "VM2": { status: "Free", user: "" }, "VM3": { status: "Free", user: "" }, "VM4": { status: "Free", user: "" }, "VM5": { status: "Free", user: "" }, "VM6": { status: "Free", user: "" }, "VM7": { status: "Free", user: "" }, "VM8": { status: "Free", user: "" }, "VM9": { status: "Free", user: "" }, "VM10": { status: "Free", user: "" } }; // Fetch the VM status from Firestore async function fetchVMStatus() { try { import { getDocs, collection } from 'https://www.gstatic.com/firebasejs/9.6.1/firebase-firestore.js'; const vmData = await getDocs(collection(db, "vmStatus")); if (!vmData.empty) { vmData.forEach(doc => { const vmName = doc.id; const vmInfo = doc.data(); defaultVMStatus[vmName] = vmInfo; }); } else { // Ensure default VM status for any missing data Object.keys(defaultVMStatus).forEach(vm => { if (!defaultVMStatus[vm]) { defaultVMStatus[vm] = { status: "Free", user: "" }; } }); } updateVMList(); } catch (error) { console.error("Error fetching VM status:", error); } } // Generate VM list in the DOM function updateVMList() { const vmListContainer = document.querySelector('.vm-list'); vmListContainer.innerHTML = ''; // Clear the current list Object.keys(defaultVMStatus).forEach(vm => { const vmElement = document.createElement('div'); const vmData = defaultVMStatus[vm]; vmElement.classList.add('vm-item'); if (vmData.status === "Free") { vmElement.classList.add('free'); } else { vmElement.classList.add('in-use'); } vmElement.innerHTML = ` ${vm}: ${vmData.status} ${vmData.status === 'In Use' ? `by ${vmData.user}` : ''} `; vmListContainer.appendChild(vmElement); }); } // Handle VM form submission (Assign VM) document.getElementById("vmForm").addEventListener("submit", async function(e) { e.preventDefault(); const selectedVM = document.getElementById("vm").value; const selectedUser = document.getElementById("name").value.trim(); if (selectedUser === "") { alert("Please select your name!"); return; } // Check if VM is free if (defaultVMStatus[selectedVM].status === "Free") { try { // Assign VM to the user defaultVMStatus[selectedVM] = { status: "In Use", user: selectedUser }; await db.collection("vmStatus").doc(selectedVM).set({ status: "In Use", user: selectedUser }); alert(`${selectedUser} has successfully assigned ${selectedVM}`); } catch (error) { console.error(`Error assigning ${selectedVM}:`, error); alert(`Error assigning ${selectedVM}. Please try again.`); } } else { alert(`${selectedVM} is already in use by ${defaultVMStatus[selectedVM].user}.`); } updateVMList(); // Update the UI after assignment document.getElementById("vmForm").reset(); // Reset the form }); // Handle VM release document.getElementById("releaseBtn").addEventListener("click", async function() { const selectedVM = document.getElementById("vm").value; const selectedUser = document.getElementById("name").value.trim(); if (selectedUser === "") { alert("Please select your name!"); return; } if (defaultVMStatus[selectedVM].status === "In Use" && defaultVMStatus[selectedVM].user === selectedUser) { try { // Release VM defaultVMStatus[selectedVM] = { status: "Free", user: "" }; await db.collection("vmStatus").doc(selectedVM).set({ status: "Free", user: "" }); alert(`${selectedUser} has successfully released ${selectedVM}`); } catch (error) { console.error(`Error releasing ${selectedVM}:`, error); alert(`Error releasing ${selectedVM}. Please try again.`); } } else { alert(`You cannot release ${selectedVM} because it is not assigned to you.`); } updateVMList(); // Update the UI after release document.getElementById("vmForm").reset(); // Reset the form }); // Fetch the initial VM status fetchVMStatus();
